        [] Uma para remover um elemento dessa lista, com base no id (evitar que ele elimine alguém caso não ache);

[] Rotas (metodos):
    [] ACESSAR,
    [] CRIAR,
    [] ATUALIZAR,
    [] EXCLUIR
